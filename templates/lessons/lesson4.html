# Ghost Game
from random import randint
print('Ghost Game')
feeling_brave = True
score = 0
while feeling_brave:
    ghost_door = randint(1, 3)
    print('Впереди 3 двери...')
    print('За одной прячется призрак')
    print('Какую дверь откроешь?')
    door = input('1-ую, 2-ую или 3-юю: ')
    door_num = int(door)
    if door_num == ghost_door:
        print('ПРИЗРАК!')
        feeling_brave = False
    else:
        print('Призрака нет!')
        print('Ты переходишь в следующую комнату.')
        score += 1
print('Беги!')
print('Игра окончена! Количество очков:', score)
