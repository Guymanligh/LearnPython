# Пример использования сложных команд
from random import randint
from time import sleep

# Работа со списками
shopping_list = ['яблоки', 'хлеб', 'молоко']
shopping_list.append('яйца')  # Добавляем элемент
print(f"Список покупок: {shopping_list}")
print(f"Всего {len(shopping_list)} товаров")

# Работа со строками
name = "Алиса"
print(f"Первая буква имени: {name[0]}")
print(f"Длина имени: {len(name)}")

# Функции
def roll_dice():
    return randint(1, 6)

def play_game():
    print("Бросаем кубик...")
    sleep(1)  # Ждём 1 секунду
    result = roll_dice()
    print(f"Выпало: {result}")
    return result

# Цикл с условием
attempts = 0
while True:
    attempts += 1
    dice = play_game()
    if dice == 6:
        print(f"Поздравляю! Нужное число выпало с {attempts} попытки!")
        break
    else:
        print("Продолжаем бросать...\n")
