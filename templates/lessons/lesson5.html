# Ghost Game → Это комментарий. Он ничего не делает.
from random import randint → Настройка команды randint для случайных чисел.
print('Ghost Game') → print печатает текст в консоли
feeling_brave = True → Создает переменную со значением True (Истина)
score = 0 → Сбрасывает счет в 0

{{ _t('l5_part2_text')|safe }}

while feeling_brave:
    ghost_door = randint(1, 3) → Выбирает случайное число от 1 до 3
    print('Впереди 3 двери...')
    print('За одной прячется призрак') → print печатает текст в консоли
    print('Какую дверь откроешь?')
    door = input('1-ую, 2-ую или 3-ью: ') → Запрашивает выбор игрока
    door_num = int(door) → Преобразует текст в число

{{ _t('l5_part3_text')|safe }}

if door_num == ghost_door:
    print('ПРИЗРАК!')     # Эта ветвь выполняется, если
    feeling_brave = False # за выбранной дверью есть призрак.
else:
    print('Призрака нет!')                       # Если призрака нет, игрок
    print('Ты переходишь в следующую комнату.')  # увидит эти сообщения
    score += 1 # Увеличивает счет на 1

{{ _t('l5_part4_text') }}

print('Беги!')
print('Игра окончена! Количество очков:', score)
