💻 Практический пример
# Пример использования основных команд
name = input("Как тебя зовут? ")  # Чтение текста
age = int(input("Сколько тебе лет? "))  # Чтение числа

# Арифметические операции
years_until_18 = 18 - age

# Условное ветвление
if years_until_18 > 0:
    print(f"{name}, тебе осталось {years_until_18} лет до совершеннолетия!")
else:
    print(f"{name}, ты уже совершеннолетний!")

# Цикл
print("Считаем до 3:")
for i in range(3):
    print(i + 1)
